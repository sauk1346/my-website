```python
import random

player_choice = input("Rock, paper or scissors (r, p, s): ").lower()

if player_choice not in ["r", "p", "s"]:
    raise Exception("Opción no válida")

computer_choice = random.choice(["r", "p", "s"])
print(f"Computer: {computer_choice}")

# Solo las combinaciones donde gana el jugador
player_wins = {("r", "s"), ("p", "r"), ("s", "p")}

if player_choice == computer_choice:
    print("Result: Draw")
elif (player_choice, computer_choice) in player_wins:
    print("Result: Player wins")
else:
    print("Result: Computer wins")
```

## 2. Ciclos

```python
import random
from time import sleep

def print_board(board):
    print("-------------")
    for i in range(3):
        print(f"| {board[i][0]} | {board[i][1]} | {board[i][2]} |")
    print("-------------")

def get_player_move(board):
    while True:
        try:
            move = input("Ingrese la fila y la columna (ej: 1,1): ")
            row, column = move.split(",")
            row = int(row.strip()) - 1
            column = int(column.strip()) - 1
            
            if 0 <= row < 3 and 0 <= column < 3 and board[row][column] == "-":
                return row, column
            else:
                print("Movimiento inválido. Intente de nuevo.")
        except (ValueError, IndexError):
            print("Formato inválido. Use el formato: fila,columna (ej: 1,1)")

def get_computer_move(board):
    # Más eficiente: obtener casillas vacías
    empty_cells = []
    for i in range(3):
        for j in range(3):
            if board[i][j] == "-":
                empty_cells.append((i, j))
    
    if empty_cells:
        return random.choice(empty_cells)
    return None

def check_winner(board):
    # Líneas horizontales
    for i in range(3):
        if board[i][0] == board[i][1] == board[i][2] != "-":
            return board[i][0]
    
    # Líneas verticales
    for i in range(3):
        if board[0][i] == board[1][i] == board[2][i] != "-":
            return board[0][i]
    
    # Diagonales
    if board[0][0] == board[1][1] == board[2][2] != "-":
        return board[0][0]
    elif board[2][0] == board[1][1] == board[0][2] != "-":
        return board[2][0]
    
    return None

def is_board_full(board):
    for row in board:
        if "-" in row:
            return False
    return True

def main():
    print("¡Bienvenido al Triki!")
    board = [["-", "-", "-"], ["-", "-", "-"], ["-", "-", "-"]]
    
    while True:
        # Turno del jugador
        print_board(board)
        row, column = get_player_move(board)
        board[row][column] = "X"
        
        winner = check_winner(board)
        if winner:
            print_board(board)
            print(f"¡{winner} gana!")
            break
        
        if is_board_full(board):
            print_board(board)
            print("¡Empate!")
            break
        
        # Turno de la computadora
        print("Turno de la computadora...")
        sleep(1)
        comp_move = get_computer_move(board)
        if comp_move:
            board[comp_move[0]][comp_move[1]] = "O"
        
        winner = check_winner(board)
        if winner:
            print_board(board)
            print(f"¡{winner} gana!")
            break
        
        if is_board_full(board):
            print_board(board)
            print("¡Empate!")
            break

if __name__ == "__main__":
    main()
```
