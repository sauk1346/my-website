$$
\begin{aligned}
    \text{Banco Chile: }\quad T_1(G) = 12 \cdot \log_{1.8} \left(\frac{G + C_i}{C_i}\right)\\[15pt]
    \text{Banco Estado:}\quad T_2(G) = 12 \cdot \log_{1.4} \left(\frac{G + C_i}{C_i}\right)
\end{aligned}
$$
