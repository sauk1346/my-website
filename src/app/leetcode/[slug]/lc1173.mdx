```python
import pandas as pd

def food_delivery(delivery: pd.DataFrame) -> pd.DataFrame:
    
    immediate_orders = (delivery['order_date'] == delivery['customer_pref_delivery_date']).sum()
    total_orders = len(delivery)
    
    immediate_percentage = round((immediate_orders / total_orders) * 100, 2)
    
    return pd.DataFrame({'immediate_percentage': [immediate_percentage]})
```
