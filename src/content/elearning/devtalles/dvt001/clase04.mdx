```js
// string
let nombre = "Peter Parker";
console.log("nombre: " + typeof nombre);

//numeros
let numero = 123;
console.log("numero: " + typeof nombre);

let numero2 = 33.0001;
console.log("numero2: " + typeof numero2);

// booleano
let esMarvel = true;
console.log("esMarvel: " + typeof esMarvel);

// undefined
let superPoder;
console.log("superPoder: " + typeof superPoder);

// null
let soyNull = null;
console.log("soyNull: " + typeof soyNull);

// symbol
let symbol1 = Symbol('a');
console.log("symbol1: " + typeof symbol1);
```
